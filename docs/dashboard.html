function loadData(isFirstLoad){ Promise.all([ fetch("data.json?ts="+Date.now()).then(r=>r.json()).catch(()=>null), fetch("insider.json?ts="+Date.now()).then(r=>r.json()).catch(()=>null), fetch("buyback.json?ts="+Date.now()).then(r=>r.json()).catch(()=>null), fetch("market.json?ts="+Date.now()).then(r=>r.json()).catch(()=>null), fetch("archive.json?ts="+Date.now()).then(r=>r.json()).catch(()=>null), ]).then(([f,i,b,mk,ar])=>{ if(f) FIRM=f; if(i) INSIDER=i; if(b) BUYBACK=b; if(mk){ LAST_MARKET=mk; } if(ar) { ARCHIVE = ar; // [핵심 수정] 실시간 새 공시(data.json)를 과거 이력(archive.json)에 병합 if (f && f.disclosures) { const merged = [...ar.disclosures, ...f.disclosures]; // 접수번호(rcept_no)를 기준으로 중복 제거 후 최신 데이터로 덮어쓰기 const unique = Array.from(new Map(merged.map(item => [item.rcept_no, item])).values()); ARCHIVE.disclosures = unique; // 운용사 목록도 병합 const allFirms = [...(ar.firms || []), ...(f.firms || [])]; ARCHIVE.firms = [...new Set(allFirms)]; } } renderTicker(LAST_MARKET); if(mk) drawChart(currentChartName); if(isFirstLoad){ if(FIRM.is_seed) $("#seedFlag").innerHTML=`${t("seed")}`; applyStaticTexts(); buildFirmTabs(); buildMarketFilter(); setFooter(); } render(); }); }
